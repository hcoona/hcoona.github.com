е°ҫйҖ’еҪ’гҖҒCPSе’Ңе°ҶйҖ’еҪ’иҪ¬еҢ–дёәеҫӘзҺҜзҡ„дёҖиҲ¬ж–№жі•

еҸ‘иЎЁдәҺ 2013-07-18 жӣҙж–°дәҺ 2020-12-25 еҲҶзұ»дәҺ Functional Programming
жң¬ж–Үд»Ӣз»ҚдәҶд»Җд№ҲжҳҜе°ҫйҖ’еҪ’пјҢд»Җд№ҲжҳҜCPSпјҢд»ҘеҸҠеҰӮдҪ•еә”з”ЁеүҚдёӨиҖ…е°ҶйҖ’еҪ’иҪ¬еҢ–дёәеҫӘзҺҜгҖӮ

жң¬ж–Үд»Ӣз»ҚдәҶд»Җд№ҲжҳҜе°ҫйҖ’еҪ’пјҢд»Җд№ҲжҳҜCPSпјҢд»ҘеҸҠеҰӮдҪ•еә”з”ЁеүҚдёӨиҖ…е°ҶйҖ’еҪ’иҪ¬еҢ–дёәеҫӘзҺҜгҖӮ

е°ҫйҖ’еҪ’ (Tail Recursion)

еңЁ Wikipedia дёӯжҳҜиҝҷж ·е®ҡд№үе°ҫйҖ’еҪ’зҡ„пјҡ

е°ҫйғЁйҖ’еҪ’жҳҜдёҖз§Қзј–зЁӢжҠҖе·§гҖӮйҖ’еҪ’еҮҪж•°жҳҜжҢҮдёҖдәӣдјҡеңЁеҮҪж•°еҶ…и°ғз”ЁиҮӘе·ұзҡ„еҮҪж•°пјҢеҰӮжһңеңЁйҖ’еҪ’еҮҪж•°дёӯпјҢйҖ’еҪ’и°ғз”Ёиҝ”еӣһзҡ„з»“жһңжҖ»иў«зӣҙжҺҘиҝ”еӣһпјҢеҲҷз§°дёәе°ҫйғЁйҖ’еҪ’гҖӮ

е°ҫйҖ’еҪ’зӣёеҜ№дәҺжҷ®йҖҡзҡ„йҖ’еҪ’жңүдёҖдёӘ_еҘҪеӨ„_ [tail-recursion-optimization]пјҡз”ұдәҺеңЁйҖ’еҪ’и°ғз”Ёз»“жқҹеҗҺжІЎжңүд»»дҪ•е…¶д»–зҡ„ж“ҚдҪңпјҢзӣҙжҺҘе°ҶйҖ’еҪ’и°ғз”Ёзҡ„з»“жһңдҪңдёәеҪ“еүҚи°ғз”Ёзҡ„з»“жһңиҝ”еӣһпјҢеӣ жӯӨдёҚйңҖиҰҒдҝқеӯҳеҪ“еүҚи°ғз”Ёж Ҳеё§пјҢеңЁйҖ’еҪ’и°ғз”Ёзҡ„ж—¶еҖҷпјҢеҸҜд»ҘзӣҙжҺҘдҝ®ж”№еҪ“еүҚж Ҳеё§дёҠзҡ„еҸҳйҮҸгҖӮ

жҖ»зҡ„жқҘиҜҙпјҢе°ҫйҖ’еҪ’еҸҜд»Ҙиў«и®Өдёәзӯүд»·дәҺеҫӘзҺҜгҖӮ

жҲ‘д»¬жқҘзңӢдёҖдёӘдҫӢеӯҗпјҡйҳ¶д№ҳеҮҪж•°гҖӮ

int factorial(int n) {
    assert (n >= 0);
    if (n == 0) {
        return 1;
    } else {
        return n * factorial(n - 1);
    }
}

иҝҷдёӘеҮҪж•°дёҚжҳҜе°ҫйҖ’еҪ’зҡ„пјҢеӣ дёәе®ғеңЁи°ғз”Ёе®ҢиҮӘе·ұеҗҺпјҢиҝҳиҰҒиҝӣиЎҢдёҖдәӣж“ҚдҪңпјҢжҜ”еҰӮиҜҙд№ҳ\(n\)гҖӮдҪҶжҳҜжҲ‘д»¬еҸҜд»ҘиҪ»жҳ“зҡ„е°Ҷе…¶ж”№еҶҷдёәе°ҫйҖ’еҪ’зҡ„пјҡ

static int factorial_recursive_aux(int acc, int n) {
    if (n) {
        return factorial_recursive_aux(acc * n, n - 1);
    } else {
        return acc;
    }
}

int factorial_recursive(int n) {
    assert (n >= 0);
    return factorial_recursive_aux(1, n);
}

йҖҡиҝҮеј•е…ҘдёҖдёӘеҸӮж•°`acc`пјҢжҲ‘д»¬еҸҜд»Ҙе°ҶеҺҹжң¬еңЁйҖ’еҪ’и°ғз”Ёд№ӢеҗҺжүҚиҝӣиЎҢзҡ„и®Ўз®—жҸҗеүҚеҲ°йҖ’еҪ’и°ғз”Ёж—¶пјҢд»ҺиҖҢе°ҶдёҖдёӘжҷ®йҖҡзҡ„йҖ’еҪ’и°ғз”ЁиҪ¬еҢ–дёәе°ҫйҖ’еҪ’и°ғз”ЁгҖӮж №жҚ®дёҠйқўжҸҗеҲ°зҡ„_дјҳеҢ–еҺҹеҲҷ_ [tail-recursion-optimization]пјҢжҲ‘д»¬иҝҳеҸҜд»Ҙиҝҷж ·дјҳеҢ–иҝҷдёҖе°ҫйҖ’еҪ’и°ғз”ЁеҮҪж•°пјҡ

static int factorial_recursive_opt(int acc, int n) {
FRO_START:
    if (n) {
        acc *= n;
        n--;
        goto FRO_START;
    } else {
        return acc;
    }
}

е®№жҳ“еҸ‘зҺ°пјҢиҝҷж ·дёҖдёӘеҮҪж•°зӯүд»·дәҺиҝҷж ·дёҖдёӘеҫӘзҺҜпјҡ

int acc = 1;
for (; n; n--) {
    acc *= n;
}

жіЁж„ҸпјҢиҷҪз„¶иҝҷйҮҢз”ЁдёҖдёӘ`while (n--)`жӣҙдёәз®ҖжҙҒпјҢдҪҶжҳҜжҲ‘иҝҳжҳҜзү№еҲ«еҶҷдәҶдёҖдёӘ`for`еҫӘзҺҜд»Ҙи·ҹдёҠйқўзҡ„д»Јз ҒдёҖдёҖеҜ№еә”гҖӮиҝҷжҳҜеӣ дёәпјҢеңЁжңүдәӣжғ…еҶөдёӢпјҢеҲӨж–ӯе’Ңжӣҙж”№еҸҳйҮҸзҡ„жӯҘйӘӨжӣҙдёәеӨҚжқӮпјҢжӯӨж—¶з”ЁдёҖдёӘ`while`еҫӘзҺҜе°ұжҜ”иҫғеӣ°йҡҫдәҶгҖӮ

еҸҜжғңе°ҶдёҖдёӘйҖ’еҪ’и°ғз”ЁиҪ¬жҚўдёәе°ҫйҖ’еҪ’и°ғз”Ёе№¶дёҚжҖ»жҳҜиҝҷд№Ҳе®№жҳ“гҖӮжҜ”еҰӮдёӢйқўиҝҷдёӘдҫӢеӯҗпјҡ

int fibonacci(int n) {
    if (n <= 0) {
        return 0;
    } else if (n == 1 || n == 2) {
        return 1;
    } else {
        return fibonacci(n - 1) + fibonacci(n - 2);
    }
}

еҶҚжҜ”еҰӮжӣҙеҠ еӨҚжқӮзҡ„еҪ’е№¶жҺ’еәҸпјҢеҝ«йҖҹжҺ’еәҸгҖӮ

еӣ дёәдёӢйқўзҡ„еҶ…е®№з”ЁеҮҪж•°ејҸиҜӯиЁҖеҸҷиҝ°жӣҙеҠ ж–№дҫҝпјҢжүҖд»ҘеңЁжӯӨе…Ҳз»ҷеҮәFibonacciеҮҪж•°еҜ№еә”зҡ„з”ЁFSharpиҜӯиЁҖеҶҷзҡ„д»Јз Ғпјҡ

let rec fibonacci n =
    match n with
    | _ when n <= 0 -> 0
    | 1 | 2 -> 1
    | _ -> (fibonacci (n - 1)) + (fibonacci (n - 2))
;;

Continuation Passing Style (CPS)

еңЁ Wikipedia дёӯжҳҜиҝҷж ·е®ҡд№үCPSзҡ„пјҡ

In functional programming, continuation-passing style (CPS) is a style of programming in which control is passed explicitly in the form of a continuation.

зҗҶи®әжҖ»жҳҜйӮЈд№ҲжҠҪиұЎпјҢжҲ‘д»¬дёҚеҰЁжқҘзңӢдёҖдёӘдҫӢеӯҗпјҡ

let rec factorial n =
    match n with
    | _ when n < 0 -> failwith "n must not be less than 0."
    | 0 -> 1
    | _ -> n * (factorial (n - 1))
;;

let factorial_cps n =
    let rec factorial_aux n cont =
        match n with
        | 0 -> cont 1
        | _ -> factorial_aux (n - 1) (fun acc -> cont (acc * n))
    in
    if n < 0 then failwith "n must not be less than 0."
    else factorial_aux n (fun x -> x)
;;

еҸҜд»ҘзңӢеҲ°пјҢжҲ‘д»¬е°ҶеҺҹжң¬дјҡеҸ‘з”ҹеңЁйҖ’еҪ’и°ғз”Ёд№ӢеҗҺиҝӣиЎҢзҡ„ж“ҚдҪңвҖ”вҖ”д№ҳ`n`пјҢйҖҡиҝҮContinuationзҡ„ж–№ејҸпјҢдҝқз•ҷеҲ°д№ӢеҗҺзҡ„жҹҗдёҖж—¶еҲ»еҶҚиҝӣиЎҢи°ғз”ЁпјҲе…·дҪ“дәҺиҝҷдёӘдҫӢеӯҗиҖҢиЁҖпјҢжҳҜеңЁ`n = 0`ж—¶пјүгҖӮе№¶дё”еңЁжҜҸж¬ЎйҖ’еҪ’и°ғз”Ёзҡ„ж—¶еҖҷпјҢйғҪе°ҶContinuationдҪңдёәеҸӮж•°д№ӢдёҖиҝӣиЎҢдј йҖ’гҖӮиҝҷе°ұжҳҜContinuation-passing styleгҖӮ

CPSзҡ„еҘҪеӨ„е°ұжҳҜпјҢиғҪеӨҹе°ҶеҺҹжң¬дёҚжҳҜе°ҫйҖ’еҪ’зҡ„жҷ®йҖҡйҖ’еҪ’и°ғз”ЁиҪ¬еҢ–дёәе°ҫйҖ’еҪ’еҪўејҸгҖӮиҮіжӯӨпјҢеҰӮдҪ•е°ҶдёҖиҲ¬йҖ’еҪ’иҪ¬еҢ–дёәеҫӘзҺҜзҡ„ж–№жі•пјҢе°ұжҜ”иҫғжҳҺжң—дәҶгҖӮ

жҺҘдёӢжқҘпјҢжҲ‘д»¬жқҘзңӢдёҖдёӘжҜ”иҫғеӨҚжқӮзҡ„_дҫӢеӯҗ_ [lambda]пјҡFibonacciгҖӮпјҲиҝҷйҮҢдёҚиҖғиҷ‘дҪҝз”Ё еҠЁжҖҒи§„еҲ’ (Dynamic Programming) еҜ№е…¶иҝӣиЎҢдјҳеҢ–зҡ„жғ…еҶөпјү

let fibonacci_cps n =
    let rec fibonacci_aux n cont =
        match n with
        | 1 | 2 -> cont 1
        | _ -> fibonacci_aux
                    (n - 1)
                    (fun acc1 -> fibonacci_aux
                                    (n - 2)
                                    (fun acc2 -> cont (acc1 + acc2)))
    in
    match n with
    | _ when n <= 0 -> 0
    | 1 | 2 -> 1
    | _ -> fibonacci_aux n (fun x -> x)
;;

еӣһйЎҫдёҖдёӢжҲ‘д»¬д№ӢеүҚеҶҷзҡ„`fibonacci`еҮҪж•°пјҢжҲ‘д»¬е®№жҳ“еҸ‘зҺ°пјҢе”ҜдёҖзҡ„дёҚеҗҢд№ӢеӨ„е°ұеңЁдәҺпјҢжҲ‘д»¬дҪҝз”Ё`fibonacci_aux`еҮҪж•°жқҘе®ҢжҲҗд»ҘеүҚ`fibonacci`еҮҪж•°зҡ„дё»иҰҒеҠҹиғҪпјҲйҖ’еҪ’пјүгҖӮиҖҢ`fibonacci_aux`еҮҪж•°дёҺеҺҹжқҘзҡ„`fibonacci`еҮҪж•°зҡ„дёҚеҗҢд№ӢеӨ„пјҢйҷӨдәҶжІЎжңүеӨ„зҗҶ`n ⇐ 0`зҡ„жғ…еҶөд»ҘеӨ–пјҢдё»иҰҒеңЁдәҺдҪҝз”ЁеҢҝеҗҚеҮҪж•°е°Ҷе®ҢжҲҗеҪ“еүҚиҜӯеҸҘд№ӢеҗҺиҰҒеҒҡзҡ„дәӢжғ…дҪңдёәеҸӮж•°иҝӣиЎҢдј йҖ’гҖӮиҝҷж ·дёҖжқҘпјҢйҖҡиҝҮе°ҶйҖ’еҪ’и°ғз”Ёд№ӢеҗҺжүҖеҒҡзҡ„ж“ҚдҪңйҖҡиҝҮеҸӮж•°иҝӣиЎҢдј йҖ’пјҢеңЁжү§иЎҢе®ҢдәҶжүҖжңүзҡ„йҖ’еҪ’жӯҘйӘӨд№ӢеҗҺпјҢеҶҚиҝӣиЎҢи®Ўз®—пјҢдҪҝеҫ—жҲ‘д»¬е°ҶдёҖдёӘеӨҚжқӮзҡ„жҷ®йҖҡйҖ’еҪ’и°ғз”ЁиҪ¬жҚўжҲҗдёәдәҶе°ҫйҖ’еҪ’и°ғз”ЁгҖӮ

еҜ№дәҺиҝҷзӮ№дёҚжҳҺзҷҪзҡ„еҗ¬дј—пјҢдёҚеҰЁжүӢе·Ҙеұ•ејҖ`n = 4`зҡ„жғ…еҶөгҖӮ

д»ҘдёӢе°Ҷfibonacci_auxз®ҖеҶҷдёәf
   f 4 (fun x -> x)
=> f 3 (fun x -> f 2 (fun y -> (fun xx -> xx) (x + y)))
=> f 3 (fun x -> f 2 (fun y -> x + y))
=> f 3 (fun x -> (fun y -> x + y) 1)
=> f 3 (fun x -> x + 1)
=> f 2 (fun x -> f 1 (fun y -> (fun xx -> xx + 1) (x + y)))
=> f 2 (fun x -> (fun y -> (fun xx -> xx + 1) (x + y)) 1)
=> f 2 (fun x -> (fun xx -> xx + 1) (x + 1))
=> (fun x -> (fun xx -> xx + 1) (x + 1)) 1
=> (fun xx -> xx + 1) 2
=> 3

йҖ’еҪ’иҪ¬дёәеҫӘзҺҜ

еҲ°иҝҷйҮҢпјҢжҲ‘д»¬иҮіе°‘зҹҘйҒ“дәҶеңЁеҮҪж•°ејҸзј–зЁӢиҜӯиЁҖдёӯпјҢеҰӮдҪ•е°ҶдёҖдёӘдёҖиҲ¬йҖ’еҪ’и°ғз”ЁиҪ¬жҚўдёәе°ҫйҖ’еҪ’и°ғз”ЁгҖӮдёӢйқўпјҢжҲ‘д»¬и®Ёи®әеңЁC++е’ҢCиҜӯиЁҖдёӯпјҢеә”иҜҘжҖҺд№ҲеҒҡгҖӮ

C++дёӯзҡ„е°қиҜ•

еҜ№дәҺC++11иҖҢиЁҖпјҢз”ұдәҺеј•е…ҘдәҶ_Lambda_ [CXX11-lambda]пјҢдҪҝеҫ—зҗҶи®әдёҠеҒҡиҝҷд»¶дәӢзҡ„йҡҫеәҰе’ҢеҮҪж•°ејҸзј–зЁӢиҜӯиЁҖе·®дёҚеӨҡгҖӮ

template<typename _FContTy>
static int fibonacci_aux(int n, _FContTy cont)
{
    if (n == 1 || n == 2) {
        return cont(1);
    } else {
        return fibonacci_aux(n - 1, [=](int acc1) {
            return fibonacci_aux(n - 2, [=](int acc2) {
                return cont(acc1 + acc2);
            });
        });
    }
}

int fibonacci(int n)
{
    return fibonacci_aux(n, [](int x) { return x; });
}

дҪҶжҳҜе®һйҷ…дёҠпјҢиҝҷж®өд»Јз Ғж— и®әжҳҜдҪҝз”Ё_жңҖж–°зүҲжң¬зҡ„Visual Studio_ [latest-vs]жҲ–иҖ…жҳҜ_жңҖж–°зүҲжң¬зҡ„G++_ [latest-gxx]йғҪж— жі•зј–иҜ‘пјҲиЎЁзҺ°дёәй•ҝж—¶й—ҙж— е“Қеә”пјҢеҚ з”ЁеҶ…еӯҳз–ҜзӢӮеўһй•ҝпјүгҖӮдҪҶжҳҜиҝҷе№¶дёҚжҳҜзј–иҜ‘еҷЁзҡ„BugпјҢиҖҢжҳҜеӣ дёәиҝҷж®өйҖ’еҪ’зЁӢеәҸеңЁзј–иҜ‘ж—¶жңҹиҪ¬жҚўжҲҗд»ҝеҮҪж•°(Functor)ж—¶дјҡдә§з”ҹдёҖдёӘж— з©·йҖ’еҪ’пјҢж„ҹе…ҙи¶Јзҡ„еҗ¬дј—жүӢе·ҘжЁЎжӢҹдёҖдёӢзј–иҜ‘еҷЁзҡ„е·ҘдҪңе°ұиғҪеҸ‘зҺ°иҝҷдёҖзӮ№гҖӮеҶөдё”е®һйҷ…дёҠжңҖеҘҪзҡ„жғ…еҶөдёӢпјҢд№ҹдјҡжңүдёӨдёӘfibonacci_auxеҮҪж•°дә§з”ҹпјҲеӣ дёәжңүдёӨдёӘдёҚеҗҢзҡ„lambdaдҪңдёәеҸӮж•°contпјүпјҢдҪҝеҫ—з”ҹжҲҗзҡ„д»Јз Ғе®һйҷ…дёҠдёҚжҳҜе°ҫйҖ’еҪ’и°ғз”ЁпјҲиҖҢжҳҜй—ҙжҺҘе°ҫйҖ’еҪ’и°ғз”ЁпјүгҖӮ

зј–иҜ‘еҷЁдёҚеӨҹиҒӘжҳҺпјҢжІЎеҠһжі•еӨ„зҗҶеҘҪиҝҷд»¶дәӢпјҢдёҚиҝҮжҲ‘д»¬еҸҜд»ҘжүӢе·ҘжқҘеӨ„зҗҶиҝҷз§Қжғ…еҶөгҖӮдҪҝз”Ёз»§жүҝе’ҢеӨҡжҖҒзҡ„зү№жҖ§пјҢжҲ‘д»¬еҸҜд»Ҙз»ҹдёҖ`cont`зҡ„зұ»еһӢгҖӮ

#include <memory>

using namespace std;

class ContF {
protected:
    virtual int Imp(int x) const = 0;
public:
    int operator()(int x) const { return Imp(x); }
};

static int f(int n, shared_ptr<ContF> cont);

class ContIdentityF : public ContF {
protected:
    virtual int Imp(int x) const override { return x; }
};

class ContInnerF : public ContF {
private:
    int x;
    shared_ptr<ContF> cont;
protected:
    virtual int Imp(int y) const override {
        return cont->operator() (x + y);
    }
public:
    ContInnerF(int x, shared_ptr<ContF> cont) : x(x), cont(cont) { }
};

class ContOuterF : public ContF {
private:
    int n;
    shared_ptr<ContF> cont;
protected:
    virtual int Imp(int x) const override {
        return f(n - 2, make_shared<ContInnerF>(x, cont));
    }
public:
    ContOuterF(int n, shared_ptr<ContF> cont) : n(n), cont(cont) { }
};

static int f(int n, shared_ptr<ContF> cont) {
    if (n == 1 || n == 2) {
        return cont->operator() (1);
    } else {
        return f(n - 1, make_shared<ContOuterF>(n, cont));
    }
}

int fibonacci(int n) {
    if (n <= 0) {
        return 0;
    } else {

#include <memory>

using namespace std;

enum class ContDataType { Identity, Outer, Inner };

struct ContData {
    const ContDataType type;
    const int x;
    const shared_ptr<ContData> p;

    ContData(ContDataType type, int x, shared_ptr<ContData> p)
        : type(type), x(x), p(p)
    {
        _ASSERT(type == ContDataType::Outer
             || type == ContDataType::Inner);
    }
    ContData()
        : type(ContDataType::Identity), x(0), p(nullptr)
    { }
};

static int f(int n, shared_ptr<ContData> data);

static int f_cont(int x, shared_ptr<ContData> data) {
    switch (data->type)
    {
    case ContDataType::Identity:
        return x;
    case ContDataType::Outer:
        return f(
            data->x - 2,
            make_shared<ContData>(ContDataType::Inner, x, data->p));
    case ContDataType::Inner:
        return f_cont(data->x + x, data->p);
    }
}

static int f(int n, shared_ptr<ContData> cont) {
    if (n == 1 || n == 2) {
        return f_cont(1, cont);
    } else {
        return f(
            n - 1,
            make_shared<ContData>(ContDataType::Outer, n, cont));
    }
}

int fib(int n) {
    if (n <= 0) {
        return 0;
    } else {
        return f(n, make_shared<ContData>());
    }
}

enum { Identity, Outer, Inner };

struct ContData {
    int type;
    int x;
    int pIdx;
};

static const int STACK_SIZE = 10000;

static int f(int n, ContData stack[], int stackTop) {
F_START:
    if (n == 1 || n == 2) {
        n = 1;
        goto CONT_START;
    } else {
        stackTop++;
        stack[stackTop].type = Outer;
        stack[stackTop].x = n;
        stack[stackTop].pIdx = stackTop - 1;
        n--;
        goto F_START;
    }

CONT_START:
    switch (stack[stackTop].type)
    {
    case Identity:
        return n;
    case Outer: {
            int nn = n;
            n = stack[stackTop].x - 2;
            stackTop++;
            stack[stackTop].type = Inner;
            stack[stackTop].x = nn;
            stack[stackTop].pIdx = stack[stackTop - 1].pIdx;
            goto F_START;
        }
    case Inner:
        n = stack[stackTop].x + n;
        stackTop = stack[stackTop].pIdx;
        goto CONT_START;
    }
}

int fib(int n) {
    if (n <= 0) {
        return 0;
    } else {
        ContData stack[STACK_SIZE];
        stack[0].type = Identity;
        return f(n, stack, 0);
    }
}
